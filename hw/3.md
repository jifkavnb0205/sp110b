```
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>
#include <string.h>
#include <sys/mman.h>

typedef int(*fn)(int);

int main() {
  char *memory = mmap(NULL,             // address
                      4096,             // size
                      PROT_READ | PROT_WRITE | PROT_EXEC,
                      MAP_PRIVATE | MAP_ANONYMOUS,
                      -1,               // fd (not used here)
                      0);               // offset (not used here)
  assert(memory != NULL);
  char code[] = {
    0x8d,0x41,0x14,
    0xc3,
    0x90,
  };

  memcpy(memory, code, 5);
  fn f = (fn) memory;

  printf("f(2,20) = %d\n", (*f)(2));
  munmap(f, 4096);
  return 0;
}
```
